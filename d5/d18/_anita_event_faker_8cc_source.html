114 UsefulAnitaEvent * AnitaEventFaker::makePureNoiseEvent(double rms, UsefulAnitaEvent * victim) const
115 {
116 // fprintf(stderr,"WARNING: makePureNoiseEvent() probably doesn't work yet\n");
117  UsefulAnitaEvent * u = victim;
118 
119  //if we don't have a model event, generate a fake time base for this one
120  if (!victim)
121  {
122  u = new UsefulAnitaEvent;
123  for (int i = 0; i < NUM_DIGITZED_CHANNELS; i++)
124  {
125  double t = 0;
126  u->fNumPoints[i] = NUM_SAMP;
127  for (int j = 0; j < NUM_SAMP; j++)
128  {
129  u->fTimes[i][j] = t;
130  double dt = 0;
131  while (dt <= 0) dt = gRandom->Gaus(1/2.6,0.09); //TODO make more general
132  t+=dt;
133  }
134  }
135  memset(u->fVolts,0,sizeof(u->fVolts));
136  }
137 
138  /* Next, we will generate thermal noise according to the instrument response, upsample it, then evaluate it at the times */
139 
140  for (int ch = 0; ch < NUM_DIGITZED_CHANNELS; ch++)
141  {
142  int chan, surf;
143  AnitaGeomTool::getSurfChanFromChanIndex(ch, surf, chan);
144  if (chan == 8) continue; //this is the clock, ignore it;
145  int ant;
146  AnitaPol::AnitaPol_t pol ;
147 
148  AnitaGeomTool::getAntPolFromSurfChan(surf,chan, ant, pol);
149 
150  //empty waveform, oversampled and a bit longer
151  AnalysisWaveform wf(360, 1/2.6, -20); //TODO, these should be configurable!
152 
153  int N = wf.Neven();
154  TGraphAligned * g = wf.updateEven();
155  for (int i = 0; i < N; i++) g->GetY()[i] = gRandom->Gaus(0, rms);
156 
157  double df = wf.deltaF();
158  FFTWComplex * G = wf.updateFreq();
159 
160  for (int i = 0; i < N/2+1; i++)
161  {
162  G[i] *= manager.response(pol, ant)->getResponse(i * df);
163  }
164 
165  //upsample a bunch
166  wf.padFreq(4);
167 
168  //evaluate at the times
169  wf.evalEven(u->fNumPoints[ch], u->fTimes[ch], u->fVolts[ch]);
170  }
171 
172  return u;
173 }

177 void AnitaEventFaker::addSignal(UsefulAnitaEvent * event, double theta, double phi, double A, double extra_delay,
178  std::complex<double> jones_H, std::complex<double> jones_V) const

183  double phi_rad = phi * TMath::DegToRad();

201  double phi0_rad = AnitaGeomTool::Instance()->getAntPhiPositionRelToAftFore(ant,pol);
202  double phi0= phi0_rad * TMath::RadToDeg();
203  double Greal = A * (pol == AnitaPol::kHorizontal ? offAxisGain_hpol : offAxisGain_vpol).Eval(phi-phi0, theta-10); //TODO: don't hardcode 10
204  std::complex<double> G = Greal * (pol == AnitaPol::kHorizontal ? jones_H : jones_V);
205 
206  double ts= (z * tan(th_rad) - R * cos(phi_rad-phi0_rad)) * 1e9 * cos(th_rad) / C_LIGHT;
207  ts+= offAxisDelay.Eval( phi-phi0, theta-10);
208  ts += extra_delay;
209 
210  for (int i = 0; i < event->fNumPoints[ich]; i++)
211  {
212 
213  if (event->fTimes[ich][i] - ts< sig_t0) continue;
214  if (event->fTimes[ich][i] - ts> sig_t1) continue;
215 
216  std::complex<double> val = G * std::complex<double> ( signal[ant][pol].evalEven(event->fTimes[ich][i] - ts),
217  signal[ant][pol].hilbertTransform()->evalEven(event->fTimes[ich][i] - ts));
218  event->fVolts[ich][i] += std::real(val);
219  }
220  }
221 }
