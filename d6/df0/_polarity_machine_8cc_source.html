348 double PolarityMachine::calculateCLEANPolarity(TGraph* gwf)
349 {
350  int max_ind = 0;
351  int min_ind = 0;
352  double max_val = 0;
353  double min_val = 0;
354  for(int i = 0; i < gwf->GetN(); i++)
355  {
356  if(gwf->GetY()[i] > max_val)
357  {
358  gwf->GetY()[i] = max_val;
359  i = max_ind;
360  }
361  if(gwf->GetY()[i] < min_val)
362  {
363  gwf->GetY()[i] = min_val;
364  i = min_ind;
365  }
366  }
367  double maxovermin = fabs(max_val)/fabs(min_val);
368  double minovermax = fabs(min_val)/fabs(max_val);
369  maxovermin = TMath::Min(maxovermin, minovermax);
370  double retval = 0;
371  if(maxovermin > .5)
372  {
373  //choose the first peak if they are similar in size
374  retval = (max_ind < min_ind) ? 1 : -1;
375  }
376  else
377  {
378  //choose the largest peak if they are far apart in size
379  retval = (fabs(max_val) < fabs(min_val)) ? -1 : 1;
380  }
381  return retval;
382 }

450 TGraph* PolarityMachine::windowWaveform(AnalysisWaveform* wf, double window_size_ns)
451 {
452  double * x = wf->even()->GetY();
453  double * xh = wf->hilbertTransform()->even()->GetY();
454  double max_dI = 0;
455  int max_ind = -1;
456  for(int i = 0; i < wf->Neven(); i++)
457  {
458  std::complex<double> X(x[i], xh[i]);
459  double dI = std::norm(X);
460  if(dI > max_dI)
461  {
462  max_dI = dI;
463  max_ind = i;
464  }
465  }
466  double prev = x[max_ind];
467  int zc_ind = -1;
468  int zc = 0;
469  for(int i = max_ind; i<wf->Neven(); i++)
470  {
471  std::complex<double> X(x[i], xh[i]);
472  double dI = std::norm(X);
473  if(prev < 0 && x[i] > 0) zc = 1;
474  if(prev > 0 && x[i] < 0) zc = 1;
475  if(zc)
476  {
477  max_dI = dI;
478  zc_ind = i;
479  break;
480  }
481  prev = x[i];
482  }
483  prev = x[max_ind];
484  int sample_diff = zc_ind - max_ind;
485  zc = 0;
486  for(int i = max_ind; i > 0; i--)
487  {
488  std::complex<double> X(x[i], xh[i]);
489  double dI = std::norm(X);
490  if(prev < 0 && x[i] > 0) zc = 1;
491  if(prev > 0 && x[i] < 0) zc = 1;
492  if(zc)
493  {
494  if(max_ind - i < sample_diff) zc_ind = i;
495  break;
496  }
497  prev = x[i];
498  }
499  int window_edge = int(ceil(window_size_ns/(2*wf->deltaT())));
500  TGraph* theOut = new TGraph(wf->even()->GetN(), wf->even()->GetX(), wf->even()->GetY());
501  double alpha = 2 * (1. - double(2 * window_edge)/double(theOut->GetN())); //tukey window alpha
502  for(int i = 0; i < theOut->GetN(); i++)
503  {
504  if(i >= (max_ind - window_edge) && i <= (max_ind + window_edge)) continue;
505  else if (i < (max_ind - window_edge))
506  {
507  double tukey_const = double(2*i)/(alpha * double(theOut->GetN()-1));
508  double coeff = .5*(1 + TMath::Cos(TMath::Pi()*(tukey_const-1)));
509  theOut->GetY()[i] *= coeff;
510  }
511  else
512  {
513  int j = 2 * max_ind - i;
514  double tukey_const = double(2*j)/(alpha * double(theOut->GetN()-1));
515  double coeff = .5*(1 + TMath::Cos(TMath::Pi()*(tukey_const-1)));
516  theOut->GetY()[i] *= coeff;
517  }
518  }
519  return theOut;
520 }

712  if(val_at_max < 0 && wf->even()->GetY()[i] > 0) break;

730  }

738  {
