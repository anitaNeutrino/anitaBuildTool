463 double PrettyAnalysisWaveform::getHighestSnr(int centreAntenna)
464 {
465  while(centreAntenna>16) centreAntenna -= 16;
466  double vpptop=TMath::MaxElement(useful->fNumPoints[centreAntenna], useful->fVolts[centreAntenna]) - TMath::MinElement(useful->fNumPoints[centreAntenna], useful->fVolts[centreAntenna]);
467  double vppmid=TMath::MaxElement(useful->fNumPoints[centreAntenna+16], useful->fVolts[centreAntenna+16]) - TMath::MinElement(useful->fNumPoints[centreAntenna+16], useful->fVolts[centreAntenna+16]);
468  double vppbot=TMath::MaxElement(useful->fNumPoints[centreAntenna+32], useful->fVolts[centreAntenna+32]) - TMath::MinElement(useful->fNumPoints[centreAntenna+32], useful->fVolts[centreAntenna+32]);
469  int nPointsRms = useful->fNumPoints[centreAntenna]/5;
470  double snrtop=0;
471  double snrmid=0;
472  double snrbot=0;
473  for(int i = 0; i < nPointsRms; i++)
474  {
475  snrtop += useful->fVolts[centreAntenna][i]*useful->fVolts[centreAntenna][i]/double(nPointsRms);
476  snrmid += useful->fVolts[centreAntenna+16][i]*useful->fVolts[centreAntenna+16][i]/double(nPointsRms);
477  snrbot += useful->fVolts[centreAntenna+32][i]*useful->fVolts[centreAntenna+32][i]/double(nPointsRms);
478  }
479  snrtop = vpptop/sqrt(snrtop);
480  snrmid = vppmid/sqrt(snrmid);
481  snrbot = vppbot/sqrt(snrbot);
482  snrtop = (snrtop>snrbot) ? snrtop:snrbot;
483  snrtop = (snrtop>snrmid) ? snrtop:snrmid;
484  return snrtop;
485 }
