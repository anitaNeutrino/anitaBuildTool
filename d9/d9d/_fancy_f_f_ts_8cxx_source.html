105 double* Acclaim::FancyFFTs::getPowerSpectrum(int len, double* input, double dt, Acclaim::FancyFFTs::conventionFlag normFlag, double* outputPtr, int threadInd){
106 
107  /*
108  Acclaim::FancyFFTs::conventionFlag determines (you guessed it) the normalization of the power spectrum.
109  For a thorough explanation see Ryan's presentation:
110  http://www.hep.ucl.ac.uk/~rjn/saltStuff/fftNormalisation.pdf
111  */
112 
113  const double ohms = 50; // Assuming 50 Ohm termination
114  double conventionNorm = 1;
115  switch (normFlag){
116  case kSum:
117  /* Sum of output powSpec == sum over V[i]*V[i] for each V[i] in input */
118  conventionNorm = 1./len;
119  break;
120 
121  case kAverage:
122  /*
123  Sum of output powSpec == sum over V[i]*V[i]/len for each V[i] in input
124  Tells you power per unit sample? ANITAns probably don't want this one.
125  */
126  conventionNorm = 1./(len*len);
127  break;
128 
129  case kTimeIntegral:
130  /*
131  Sum of output powSpec == sum over dt*V[i]*V[i] for each V[i] in input
132  Tell you total time integrated power.
133  */
134  conventionNorm = dt/len;
135  break;
136 
137 
138  case kPowSpecDensity:
139  /*
140  Sum of df*psd[i] for each psd[i] in output == sum over dt*V[i]*V[i]/df for each V[i] in input
141  == sum over N*dt*dt*V[i]*V[i] for each V[i] in input
142  Makes this function return the "power spectral density".
143  i.e. frequency bins are normalized such that zero padding the waveform won't affect bin content.
144  */
145 
146  conventionNorm = dt*dt; /* since df = 1./(len*dt) */
147  break;
148 
149 
150  case kPowSpecDensity_dBm:
151  /*
152  Sum of df*psd[i] for each psd[i] in output == sum over dt*V[i]*V[i]/df/ohms for each V[i] in input
153  == sum over N*dt*dt*V[i]*V[i]/ohms for each V[i] in input
154  Makes this function return the "power spectral density" (assuming 50 Ohm termination).
155  i.e. frequency bins are normalized such that zero padding the waveform won't affect bin content.
156  */
157 
158  conventionNorm = dt*dt/ohms; /* since df = 1./(len*dt) */
159  break;
160 
161  default:
162  /* You shouldn't get here and now I'm going to tell you that. */
163  std::cerr << "Invalid Acclaim::FancyFFTs::conventionFlag in Acclaim::FancyFFTs::getPowerSpectrum(int len, double* input, double dt, Acclaim::FancyFFTs::conventionFlag normFlag) " << std::endl;
164  }
165 
166 
167  /*
168  Do FFT without putting the output in a new array.
169  we need to normalize the output so lets do that when we move it.
170  */
171  doFFT(len, input, false, threadInd);
172 
173 
174  std::pair<int, int> key(len, threadInd);
175 
176  /* Get the fftw_malloc'd array that the plan uses */
177  complex<double>* rawFftOutputPtr = (complex<double>*) fComplex[key];
178 
179  const int powSpecLen = getNumFreqs(len);
180 
181  double* powSpec = NULL;
182  if(outputPtr==NULL){
183  powSpec = new double[powSpecLen];
184  }
185  else{
186  powSpec = outputPtr;
187  }
188 
189  /* Pedantry with the normalization (only half the nyquist bin if there's an even number of bins) */
190  bool halfPowerNquist = (len%2)==0 ? true : false;
191 
192  for(int i=0; i<powSpecLen; i++){
193 
194  /* Take account of symetry in normalization of power spectrum */
195  double symmetryFactor = 2;
196  if(i==0 || (halfPowerNquist && i==powSpecLen-1)){
197  symmetryFactor = 1;
198  }
199 
200  powSpec[i] = symmetryFactor*conventionNorm*(pow(std::abs(rawFftOutputPtr[i]), 2));
201  }
202  return powSpec;
203 }

466 complex<double>* Acclaim::FancyFFTs::zeroPadFFT(complex<double>* fft, complex<double>* output, int numSamples, int numSamplesUpsampled){
467 
468  const int numFreqs = getNumFreqs(numSamples);
469  const int numFreqsPadded = getNumFreqs(numSamplesUpsampled);
470 
471  complex<double>* fftPadded = NULL;
472  if(output==NULL){
473  fftPadded = new complex<double>[numFreqsPadded];
474  }
475  else{
476  fftPadded = output;
477  }
478 
479  // this is the mistake!!!!
480  // Double_t scale = numFreqsPadded/numFreqs;
481 
482  // Here I scale the padded FFT so that it is as if I fourier transformed a longer waveform.
483  // (There is a scale factor of length picked up from a forward FFT.)
484  Double_t scale = numSamplesUpsampled/numSamples;
485 
486  for(int freqInd=0; freqInd<numFreqs; freqInd++){
487  fftPadded[freqInd].real(fft[freqInd].real()*scale);
488  fftPadded[freqInd].imag(fft[freqInd].imag()*scale);
489  }
490  for(int freqInd=numFreqs; freqInd<numFreqsPadded; freqInd++){
491  fftPadded[freqInd] = 0;
492  }
493

495  // to the other bins (due to them containing the negative frequency components)
496  const Double_t sqrtHalf = TMath::Sqrt(0.5);
497  fftPadded[numFreqs-1].real(sqrtHalf*fftPadded[numFreqs-1].real());
498  fftPadded[numFreqs-1].imag(sqrtHalf*fftPadded[numFreqs-1].imag());

550 double* Acclaim::FancyFFTs::crossCorrelate(int len, double* v1, double* v2, int threadInd){
551  /*
552  Cross correlation is the same as bin-by-bin multiplication in the frequency domain (with a conjugation).
553  Will assume lengths are the same for now.
554  */
555 
556  /* Store output of FFT2 in tempVals */
557  complex<double>* tempVals2 = doFFT(len, v2, true, threadInd);
558 
559  /* Leave output of FFT1 in internal arrays to avoid an unnecessary copy */
560  doFFT(len, v1, false, threadInd);
561 
562  std::pair<int, int> key(len, threadInd);
563 
564  /* Get pointer to internal array */
565  complex<double>* tempVals1 = (complex<double>*) fComplex[key];
566 
567  /* Take the product */
568  int numFreqs = getNumFreqs(len);
569  for(int i=0; i<numFreqs; i++){
570  tempVals1[i] *= std::conj(tempVals2[i]);
571  }
572 
573  delete [] tempVals2;
574 
575  /* Product back to time domain */
576  double* crossCorr = doInvFFT(len, tempVals1, true, threadInd);
577 
578  /*
579  Picked up two factors of len when doing forward FFT, only removed one doing invFFT.
580  This takes out the second factor.
581  */
582  for(int i=0; i<len; i++){
583  crossCorr[i] /= len;
584  }
585 
586  return crossCorr;
587 }

666 double* Acclaim::FancyFFTs::crossCorrelatePadded(int len, int padFactor, complex<double>* fft1, complex<double>* fft2,
667  double* output, bool copyOutputToNewArray, int threadInd, bool doNormalization){
668 
669 
670  /*
671  Cross correlation is the same as bin-by-bin multiplication (and some conjugation) in the frequency domain.
672  Will assume lengths are the same for now.
673  */
674 
675 
676  /* Stops tempVals returning NULL, normally done in doFFT step.
677  But the nice thing about this class is it means that I won't be duplicating work. */
678  int padLen = len*padFactor;
679  makeNewPlanIfNeeded(padLen, threadInd);
680 
681  std::pair<int, int> key(padLen, threadInd);
682 
683  // Grab array associated with plan from internal memory
684  complex<double>* tempVals = (complex<double>*) fComplex[key];
685 
686  /* Take the product */
687  // After profiling, it turns out this complex multiplication is slow (the operator* makes it look so cheap!)
688  // We only need to do this expensive inner loop multiplication for the non-padded frequency bins
689  // since everything else is = 0;
690  int numFreqs = getNumFreqs(len);
691  for(int i=0; i<numFreqs; i++){
692  tempVals[i] = fft1[i]*std::conj(fft2[i]);
693  }
694  int numPadFreqs = getNumFreqs(padLen);
695  for(int i=numFreqs; i < numPadFreqs; i++){
696  tempVals[i] = 0;
697  }
698 
699  /* Product back to time domain */
700  double* crossCorr = output;
701  if(crossCorr==NULL){
702  /* Allocates new memory */
703  crossCorr = doInvFFT(padLen, tempVals, copyOutputToNewArray, threadInd, doNormalization);
704  }
705  else{
706  /* Does not allocate new memory */
707  crossCorr = doInvFFT(padLen, tempVals, crossCorr, copyOutputToNewArray, threadInd, doNormalization);
708  }
709 
710  if(doNormalization){
711  /*
712  Picked up two factors of len when doing forward FFT, only removed one doing invFFT.
713  This takes out the second factor.
714 
715  This is now a non-trivial normalization.
716  Picked up two factors of len when doing forward FFTs, removed a factor of padLen when doing inverse FFT
717  This takes out the second factor.
718  */
719 
720  // std::cout << doNormalization << std::endl;
721  double normalization = double(padLen)/(len*len);
722  for(int i=0; i<padLen; i++){
723  // std::cout << crossCorr[i] << std::endl;
724  crossCorr[i] *= normalization;
725  }
726  }
727 
728  return crossCorr;
729 }

768 void Acclaim::FancyFFTs::normalizeFourierDomain(int n, std::complex<double>* fft){
769 
770  const int numFreqs = getNumFreqs(n);
771  Double_t sumOfVSquared = 0;
772  for(int freqInd=0; freqInd < numFreqs; freqInd++){
773  Double_t re = fft[freqInd].real();
774  Double_t im = fft[freqInd].imag();
775  Double_t factor = freqInd == numFreqs - 1 ? 0.5 : 1;
776  sumOfVSquared += factor*(re*re + im*im);
777  }
778 
779  Double_t timeDomainVSquared = sumOfVSquared/(n*n/2);
780  Double_t norm = TMath::Sqrt(timeDomainVSquared);
781 
782  for(int freqInd=0; freqInd < numFreqs; freqInd++){
783  Double_t re = fft[freqInd].real();
784  Double_t im = fft[freqInd].imag();
785 
786  fft[freqInd].real(re/norm);
787  fft[freqInd].imag(im/norm);
788  }
