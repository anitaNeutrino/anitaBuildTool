123 
124  if (!fJustInitialized) rmsFifoPos++;
125  if (rmsFifoPos >= fifoLength) {
126  rmsFifoPos = 0;
127  rmsFifoFillFlag = true;
128  }
129 
130 
131  for (int phi=0; phi<NUM_PHI; phi++) {
132  for (int ringi=0; (AnitaRing::AnitaRing_t)ringi != AnitaRing::kNotARing; ringi++) {
133  AnitaRing::AnitaRing_t ring = (AnitaRing::AnitaRing_t)ringi;
134  for (int poli=0; (AnitaPol::AnitaPol_t)poli != AnitaPol::kNotAPol; poli++) {
135  AnitaPol::AnitaPol_t pol = (AnitaPol::AnitaPol_t)poli;
136 
137 
138  //subtract fifo position that is expiring (if there is one)
139  if (rmsFifoFillFlag) {
140  double valueSub = rmsFifo[rmsFifoIndex(phi,ringi,poli,rmsFifoPos)];
141  rmsAvg[phi][ringi][poli] -= valueSub;
142  }
143 
144  //get the rms
145  const TGraphAligned *currWave = filtered->getFilteredGraph(phi,ring,pol)->even();
146  double value = pow(currWave->GetRMS(2),2)/fifoLength;
147 
148  //add the new fifo position to the running average
149  rmsAvg[phi][ringi][poli] += value;
150 
151  //save it for the fifo for when you gotta subtract it later
152  rmsFifo[rmsFifoIndex(phi,ringi,poli,rmsFifoPos)] = value;
153 
154  }
155  }
156  }
157 
158  return;
159 }
