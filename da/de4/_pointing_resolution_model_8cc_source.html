52 double * UCorrelator::PointingResolution::computeProbabilityDensity(int N,
53  const double * __restrict__ vphi,
54  const double * __restrict__ vtheta,
55  double * __restrict__ p)
56 {
57  if (!p) p = new double[N];
58 
59  //TODO vectorize the first part of this loop
60 
61  int start = 0;
62 
63 #pragma omp simd
64  for (int i = start; i < N; i++)
65  {
66  double phidiff = vphi[i]-phi;
67  double thetadiff = vtheta[i]-theta;
68  double z= ( phidiff * phidiff * inv_dphi2) -
69  (phidiff * thetadiff * inv_dphidtheta) +
70  (thetadiff*thetadiff * inv_dtheta2);
71  z *= expterm;
72  p[i] = norm * exp(z);
73  }
