468 double Y::Getyweight(double pnu, double y, int nu_nubar, int currentint){
469  //from Connolly Calc 2011, Equations 9, 10, 11, 16, and 17.
470  // double dy=0.;//default
471  //Ev, cc or nc, nu or nubar.
472 
473  double C0_highbar, C0_lowbar,C0_high, C0_low;//these C0's are normalization factors.
474  double dNdy=0.;//default
475  double U, W, B, T;//are added in to help with readability of equations.
476  double C1_low, C2, C1_high;
477  double weighty;
478  double epsilon=log10(pnu/1.E9);
479 
480  C2=fC2->Eval(epsilon);//Eq(17)
481  C1_low=fC1_low->Eval(epsilon);//Eq(16) (Low region)
482 
483  C1_high=fC1_high[nu_nubar][currentint]->Eval(epsilon);//Eq(16)(High region)
484 
485 
486  if(nu_nubar==0) {
487  U=1-1/C2;
488  W=fabs( (ymax_high-C1_high)/(ymin_high-C1_high));
489  B=(pow(ymax_low-C1_low, 1/C2)/(ymax_low-C1_high));
490  T=B*((pow(ymax_low-C1_low, U)-pow(ymin_low-C1_low, U) )/U)+log(W);
491  C0_high=1/T;
492  C0_low=C0_high*(pow(ymax_low-C1_low, 1/C2))/(ymax_low-C1_high);
493 
494  if(y<ymax_low){//Eq(9)
495  // dy=0.00002;
496  dNdy=C0_low/pow(y-C1_low, 1/C2);//Eq(10)
497  }
498  else if(y>=ymax_low && y<1.){//Eq(9)
499  // dy=0.001;
500  dNdy=C0_high/(y-C1_high);//Eq(10)
501  }
502  else{
503  dNdy=0.;
504  cout<<"y value is outside of the domain of y.\n";
505  }
506  }
507  else if(nu_nubar==1){
508  U=1-1/C2;
509  W=fabs( (ymax_high-C1_high)/(ymin_high-C1_high));
510  B=(pow(ymax_low-C1_low, 1/C2)/(ymax_low-C1_high));
511  T=B*((pow(ymax_low-C1_low, U)-pow(ymin_low-C1_low, U) )/U)+log(W);
512  C0_highbar=1/T;
513  C0_lowbar=C0_highbar*(pow(ymax_low-C1_low, 1/C2))/(ymax_low-C1_high);
514 
515  if(y<ymax_low){
516  // dy=0.00002;
517  dNdy=C0_lowbar/pow(y-C1_low, 1/C2);
518  }
519  else if(y>=ymax_low && y<1.){
520  // dy=0.001;
521  dNdy=C0_highbar/(y-C1_high);
522  }
523  else{
524  dNdy=0;
525  cout<<"y value is outside of the domain of y.\n";
526  }
527  }
528  else{
529  cout<<"Nu_nubar is not defined!\n";
530  }
531  weighty=dNdy;
532  return weighty;
533 }//Getyweight
