310 Double_t AveragedSurfHk::getRFPowerInK(int surf, int chan)
311 {
312  if(surf<0 || surf>=ACTIVE_SURFS)
313  return -1;
314  if(chan<0 || chan>=RFCHAN_PER_SURF)
315  return -1;
316  Int_t adc=avgRFPower[surf][chan];
317 
318  Double_t kelvin=AnitaEventCalibrator::Instance()->convertRfPowToKelvin(surf,chan,adc);
319  if(TMath::IsNaN(kelvin)) return 290;
320  if(kelvin<0) return 290;
321  if(kelvin>1e6) return 290;
322  return kelvin;

327 Double_t AveragedSurfHk::getRMSRFPowerInK(int surf, int chan)
328 {
329  if(surf<0 || surf>=ACTIVE_SURFS)
330  return -1;
331  if(chan<0 || chan>=RFCHAN_PER_SURF)
332  return -1;
333  Int_t adcLow=avgRFPower[surf][chan]-rmsRFPower[surf][chan];
334  Int_t adcHigh=avgRFPower[surf][chan]+rmsRFPower[surf][chan];
335  Double_t kelvinHigh=AnitaEventCalibrator::Instance()->convertRfPowToKelvin(surf,chan,adcHigh);
336  Double_t kelvinLow=AnitaEventCalibrator::Instance()->convertRfPowToKelvin(surf,chan,adcLow);
337  if(TMath::IsNaN(kelvinHigh)) return 290;
338  if(kelvinHigh<0) return 290;
339  if(kelvinHigh>1e6) return 290;
340  if(TMath::IsNaN(kelvinLow)) return 290;
341  if(kelvinLow<0) return 290;
342  if(kelvinLow>1e6) return 290;
343  return (kelvinHigh-kelvinLow)/2;
344 
345 }
