206 void DeglitchFilter::processOne(AnalysisWaveform * wf, const RawAnitaHeader * header, int ant, int pol)
207 {
208 
209  TGraphAligned * g = action == DELETE ? wf->updateUneven() : wf->updateEven();
210 
211  //circular buffer of values, initialize with first
212  int ndelete = 0;
213  int delete_list[max_remove]; // only delete up to 10 values
214 
215  for (int i = 0; i < g->GetN(); i++)
216  {
217  double max = 0;
218 
219  if (i > 0)
220  {
221  int start = TMath::Max(0, i - neighbors);
222  int end = i-1;
223  max = max_abs(end-start+1, g->GetY() + start);
224  }
225 
226  if (i < g->GetN()-1)
227  {
228  int start = i+1;
229  int end = TMath::Min(g->GetN()-1, i+neighbors);
230  max = TMath::Max(max,max_abs(end-start+1, g->GetY() + start));
231  }
232 
233  if (g->GetY()[i] > max + thresh)
234  {
235  if (ndelete > max_remove)
236  {
237  fprintf(stderr,"DeglitchFilter: ALREADY REMOVED MORE THAN %d points in his waveform. Giving up.\n", max_remove);
238  return;
239  }
240 
241  delete_list[ndelete++] = i;
242  }
243  }
244 
245  nremoved+=ndelete;
246 
247  for (int i = 0; i < ndelete; i++)
248  {
249  if (action == DELETE)
250  {
251  g->RemovePoint(delete_list[i]-ndelete);
252  }
253  else
254  {
255  g->GetY()[delete_list[i]] = delete_list[i] == 0 ? g->GetY()[1] :
256  delete_list[i] == g->GetN()-1 ? g->GetY()[g->GetN()-2] :
257  0.5 * g->GetY()[delete_list[i]-1] + 0.5 * g->GetY()[delete_list[i] +1];
258  }
259  }

263 void IFFTDiffFilter::processOne(AnalysisWaveform* g, const RawAnitaHeader * header, int ant, int pol) {
264 
265  int nfreq = g -> Nfreq();
266 
267  double dw = 2 * M_PI * g -> deltaF();
268  double cosOrder = cos(order * (0.5 + 2 * branchOrder) * M_PI); // i^x = cos(x * (0.5 + 2 * k) * pi) + i * sin(x * (0.5 + 2 * k) * pi), k = ..., -1, 0, 1, ...
269  double sinOrder = sin(order * (0.5 + 2 * branchOrder) * M_PI);
270 
271  FFTWComplex * vals = g -> updateFreq();
272  vals[0] = order ? 0 : vals[0]; // Just in case the FFT itself (order == 0) is desired. But should be zero, anyway.
273  for (int i = 1; i < nfreq; i++) {
274 
275  double wOrder = pow(i * dw, order);
276  vals[i].re = wOrder * (cosOrder * vals[i].re - sinOrder * vals[i].im);
277  vals[i].im = wOrder * (sinOrder * vals[i].re + cosOrder * vals[i].im);
278  }
279  if (fmod(order, 2) && !(nfreq % 2)) vals[nfreq - 1] = 0; // "For odd order and even [nfreq], the Nyquist mode is taken zero."
280
