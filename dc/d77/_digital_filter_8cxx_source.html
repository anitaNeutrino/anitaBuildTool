269 void FFTtools::TransformedZPKFilter::transform(FilterTopology type , double w, double dw)
270 {
271 
272  std::complex<double> W=tan(TMath::Pi()*w/2); // prewarp frequencies
273  std::complex<double> Wh = tan(TMath::Pi()*(w+dw)/2);
274  std::complex<double> Wl = tan(TMath::Pi()*(w-dw)/2);
275  std::complex<double> dW = (Wh - Wl)/2. ;
276 
277  std::vector<std::complex<double> > new_zeroes;
278  std::vector<std::complex<double> > new_poles;
279  std::complex<double> new_gain = gain;
280 
281  if (type == LOWPASS)
282  {
283  for (size_t i = 0; i < zeroes.size(); i++)
284  {
285  new_zeroes.push_back(zeroes[i] * W);
286  new_gain /=W;
287  }
288 
289  for (size_t i = 0; i < poles.size(); i++)
290  {
291  new_poles.push_back(poles[i] * W);
292  new_gain *=W;
293  }
294  }
295  else if (type == HIGHPASS)
296  {
297  for (size_t i = 0; i < zeroes.size(); i++)
298  {
299  new_zeroes.push_back(W/zeroes[i]);
300  new_gain *=-zeroes[i];
301  }
302 
303  //extend zeroes to poles, if necessary
304  for (size_t i = zeroes.size(); i < poles.size(); i++)
305  {
306  new_zeroes.push_back(0);
307  }
308 
309 
310  for (size_t i = 0; i < poles.size(); i++)
311  {
312  new_poles.push_back(W/poles[i]);
313  new_gain /= -poles[i];
314  }
315 
316  //extend poles to zeroes, if necessary
317  for (size_t i = poles.size(); i < zeroes.size(); i++)
318  {
319  new_poles.push_back(0);
320  }
321 
322  }
323  else
324  {
325  assert(dW!=0.);
326  if (type == BANDPASS)
327  {
328  for (size_t i = 0; i < zeroes.size(); i++)
329  {
330  std::complex<double> b = zeroes[i] * dW;
331  std::complex<double> x = sqrt(b*b - Wh*Wl);
332  new_zeroes.push_back(b + x);
333  new_zeroes.push_back(b - x);
334  new_gain *= 1./(2.*dW);
335  }
336  for (size_t i = zeroes.size(); i < poles.size(); i++)
337  {
338  new_zeroes.push_back(0);
339  }
340 
341  for (size_t i = 0; i < poles.size(); i++)
342  {
343  std::complex<double> b = poles[i] * dW;
344  std::complex<double> x = sqrt(b*b - Wh*Wl);
345  new_poles.push_back(b + x);
346  new_poles.push_back(b - x);
347  new_gain *= 2.*dW;
348  }
349 
350  for (size_t i = poles.size(); i < zeroes.size(); i++)
351  {
352  new_poles.push_back(0);
353  }
354  }
355  else if (type == NOTCH)
356  {
357  std::complex<double> extra = sqrt(std::complex<double>(-Wh*Wl));
358 
359  for (size_t i = 0; i < zeroes.size(); i++)
360  {
361  std::complex<double> b =dW / zeroes[i];
362  std::complex<double> x = sqrt(b*b - Wh*Wl);
363  new_zeroes.push_back(b + x);
364  new_zeroes.push_back(b - x);
365  new_gain *= -zeroes[i];
366  }
367  for (size_t i = zeroes.size(); i < poles.size(); i++)
368  {
369  new_zeroes.push_back(extra);
370  new_zeroes.push_back(-extra);
371  }
372 
373  for (size_t i = 0; i < poles.size(); i++)
374  {
375  std::complex<double> b = dW / poles[i];
376  std::complex<double> x = sqrt(b*b - Wh*Wl);
377  new_poles.push_back(b + x);
378  new_poles.push_back(b - x);
379  new_gain /= -poles[i];
380  }
381 
382  for (size_t i = poles.size(); i < zeroes.size(); i++)
383  {
384  new_poles.push_back(extra);
385  new_poles.push_back(-extra);
386  }
387  }
388  }
389 
390 
391  gain = std::real(new_gain);
392  zeroes = new_zeroes;
393  poles = new_poles;

401

443 static void computeSavitzkyGolayCoefficients(double *x, int m, int nl, int nr, int deriv = 0)
444 {
445  assert(nl >= 0 && nr >= 0 && m >= 0 && m > deriv);
446  int size = nl+nr+1;
447 
448  //just use ROOT TMatrix here... performance is not crucial
449  TMatrixD M(size,m+1);
450  for (int i = 0; i < size; i++)
451  {
452  for (int j = 0; j <= m; j++)
453  {
454  M(i,j) = TMath::Power(i-nl, j);
455  }
456  }
457  TMatrixD Mcopy(M);
458  TDecompLU lu(Mcopy.T() * M);
459  TVectorD y(m+1);
460  y(deriv) = 1;
461  lu.Solve(y);
462 
463 
464  double factorial = TMath::Factorial(deriv);
465  for (int i = 0; i < size; i++)
466  {
467  double dotproduct = y(0);
468  for (int j = 1; j <=m; j++)
469  {
470  dotproduct += y(j) * M(i,j);
471  }
472  x[i] = dotproduct*factorial;
473  }
474 }

808 static void reflectRoots(int N, const std::complex<double> * roots, std::vector<std::complex<double> > & output, double eps)
809 {

811

813  {
814  double M = abs(roots[i]);
815  if ( M-1 < -1e-9) // woohoo
816  {
817  output[i] = roots[i];

819  else if (M-1 > 1e-9) //reflect it

821  output[i] = 1./roots[i];

826  }

901 FFTtools::ThiranFilter::ThiranFilter(double delay, int order)
902 {
903  //implementation based on https://ccrma.stanford.edu/~jos/pasp/Thiran_Allpass_Interpolators.html
904 
905  if (delay < 0)
906  {
907 
908  fprintf(stderr,"ThiranFilter requires non-negative delay\n");
909  order = 0;
910  }
911 
912  if (order < 1)
913  {
914  order = ceil(delay);
915  }
916 
917  int N = order; //to make code more readable
918 
919  acoeffs.push_back(1);
920  for (int k = 1; k <= N; k++)
921  {
922  double a_k = ( (k % 2) ? -1 : 1) * TMath::Binomial(N,k);
923 
924  for (int n = 0; n <= N; n++)
925  {
926  a_k *= (delay - N + n);
927  a_k /= (delay - N + k + n);
928  }
929 
930  acoeffs.push_back(a_k);
931  }
932 
933  //the bcoeffs will be the acoeffs backwards
934 
935  for (int i = N; i >=0; i--)
936  {
937  bcoeffs.push_back(acoeffs[i]);
938  }
