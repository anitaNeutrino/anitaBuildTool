251  std::vector<std::pair<int,double> > * max_dens, double * inv_two_pi_sqrt_det ) const
252 {
253  std::vector<std::pair<int,double> > segs;
254 
255  double inv = computeContributions(sum,pat,pol,peak,segs,bases,0,max_dens);
256  int N = segs.size();
257  double answer = 0;
258 
259 
260  double invnorm = 1;
261 
262  if (normalized)
263  {
264  double norm = 0;
265  for (int i =0; i< N; i++)
266  {
267  norm += segs[i].second;
268  }
269  if (!inv || norm < p.min_p_on_continent)
270  {
271  printf("below minimum: %g!\n", norm);
272  return -1; // there can be no overlap
273  }
274  invnorm = 1./norm;
275  }
276 
277  if (inv_two_pi_sqrt_det) *inv_two_pi_sqrt_det = inv;
278 
279 
280  //pick the right thing to overlap with
281  const double * the_rest = mode == OVERLAP_SUM_SQRTS ? getProbSqrtSums(normalized) :
282  mode == OVERLAP_SQRT_SUMS || mode == OVERLAP_SUMS ? getProbSums(normalized) :
283  getProbMaxes(normalized);
284 
285  for (int i =0; i< N; i++)
286  {
287  int seg = segs[i].first;
288 
289  double p_this = weight*segs[i].second * invnorm;
290 
291  double p_other = the_rest[seg];
292 
293 
294  if (remove_self && mode == OVERLAP_SQRT_SUMS)
295  {
296  p_other -= p_this;
297  }
298 
299  if (remove_self && mode == OVERLAP_SUM_SQRTS)
300  {
301  p_other -= sqrt(p_this); //this might help with floating point precision to do first... maybe.
302  }
303 
304  if (remove_self && mode == OVERLAP_MAXES && p_this == p_other) p_other = normalized ? max2_ps_norm[seg] : max2_ps[seg];
305 
306  double danswer = (mode == OVERLAP_SUM_SQRTS ? sqrt(p_this) : p_this) * p_other;
307 
308  if (mode == OVERLAP_SQRT_SUMS || mode == OVERLAP_MAXES) danswer = sqrt(danswer);
309 
310  if (remove_self && mode == OVERLAP_SUMS)
311  {
312  danswer -= p_this;
313  }
314 
315  answer += danswer;
316 
317  }
318 
319  if (segments) *segments = segs;
320 
321  // avoid negative roundoff
322  if (answer < 0) answer = 0;
323  return answer;
